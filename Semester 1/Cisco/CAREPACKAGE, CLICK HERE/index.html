Router – Basic Configuration

  ----Instellen Hostname----

router#configure terminal                                         conf t
router(config)#hostname R1

R1# no hostname                                                   hostname weghalen

----Instellen remote controle----

R1(config)# banner login “Router van het departement”
R1(config)# banner motd “Verboden toegang onbevoegd personeel!”

R1(config)# enable password pxl                                   zonder encryptie
R1(config)# enable secret pxl                                     met encryptie



R1(config)# line console 0
R1(config-line)# password pxl
R1(config-line)# login
R1(config-line)# logging synchronous                              (unsolited messages)

R1(config)# line vty 0 15                                         vty (0-15) virtual teletype terminal
R1(config-line)# password pxl
R1(config-line)# login
R1(config-line)# logging synchronous
R1(config-line)# exit

R1(config)# service password-encryption                            password is zonder en slaat die met encryptie op

Instellen Ethernet/Seriële/Loopback interface

  ----Instellen Ethernet----

R1configure terminal
R1(config)#interface gigabitEthernet 0/0
R1(config-if)#description Connectie met LAN1
R1(config-if)#ip address 192.168.1.1 255.255.255.0
R1(config-if)#no shutdown
R1(config-if)#exit
R1(config)#interface gigabitEthernet 0/1
R1(config-if)#description Connectie met LAN2
R1(config-if)#ip address 192.168.2.1 255.255.255.0
R1(config-if)#no shutdown

 ----Instellen Serieël-----

R1(config)#interface serial 0/0/0
R1(config-if)#description Connectie naar R2
R1(config-if)#ip address 200.200.200.2 255.255.255.252
R1(config-if)#clock rate 128000
R1(config-if)#no shutdown

 ----Instellen Loopback-----                                          localhost

R1(config)#interface loopback 0
R1(config-if)#ip address 172.0.0.1 255.255.255.255
R2(config)#interface loopback 0
R2(config-if)#ip address 172.0.0.2 255.255.255.25

=============Border_Router#show run
hostname Border_Router

interface FastEthernet0/0
 no ip address
 duplex auto
 speed auto
 shutdown
!
interface FastEthernet0/1				-> toekenen ip
 ip address 20.20.20.2 255.255.255.0
 ip nat outside
 duplex auto
 speed auto
!
interface Serial0/0/0				-> toekenen ip
 ip address 192.168.0.2 255.255.255.252
 ip nat inside
!
interface Serial0/0/1
 no ip address
 clock rate 2000000
 shutdown
!
interface Serial0/1/0				-> toekenen ip
 ip address 192.168.0.5 255.255.255.252
 ip nat inside
 clock rate 64000
!
interface Serial0/1/1
 no ip address
 clock rate 2000000
 shutdown
!
interface Vlan1
 no ip address
 shutdown
!
router rip
 version 2
 passive-interface FastEthernet0/1
 network 192.168.0.0
 default-information originate      -> naar buiten
 no auto-summary
!
ip nat inside source list 10 interface FastEthernet0/1 overload       -> acl 10
ip nat inside source static 172.16.0.2 20.20.20.3 
ip classless
ip route 0.0.0.0 0.0.0.0 FastEthernet0/1 							-> naar buiten va fasteeth 1
!
ip flow-export version 9
!
!
access-list 10 permit 10.0.0.0 0.0.1.255				-> ACL 10 laat alles door van 10.0.0.0 + wildcard

end
