//Socket --> port based address translation (PAT)
//IP:Poort


static PAT:
en
conf t
ip nat inside source static tcp 10.0.0.2 443 50.0.0.2 443 
// ip nat inside source static [protocol] [intern ip] [poort] [extern ip] [poort]

dynamic PAT{
	en
	conf t

	//Optie 1{
		//Maak ACL aan, moet toegelaten hosts/denied hosts:
		access-list 1 permit 10.0.0.10 0.0.0.0
		access-list 1 permit 10.0.0.20 0.0.0.0
		access-list 1 deny any

		//NAT pool maken:
		ip nat pool WWW 50.0.0.3 50.0.0.4 netmask 255.255.255.0
		//dus --> ip nat pool [pool naam] [begin ip] [eind ip] netmask [subnetmask]

		//Pool toevoegen aan ACL
		ip nat inside source list 1 pool WWW overload
		//dus --> ip nat inside source list [ACL nummer] pool [pool naam] overload

		//dan gaan we naar de interne interface:
		int g0/0
		ip nat inside

		//dan naar de externe interface:
		int g0/1 
		ip nat outside
		
		
		//Voorbeeld{
			enable
			configure terminal
			Enter configuration commands, one per line. End with CNTL/Z.
			access-list 1 permit 10.0.0.10 0.0.0.0
			access-list 1 permit 10.0.0.20 0.0.0.0
			access-list 1 deny any
			ip nat pool ccna 50.0.0.1 50.0.0.1 netmask 255.0.0.0
			ip nat inside source list 1 pool ccna overload
			interface FastEthernet 0/0
			ip nat inside
			exit
			interface Serial 0/0/0
			ip nat outside
			exit
			
		}
	}
}
